```ts
// src/lib/shell/calculator-store.svelte.ts
export interface CalculatorStoreOptions<T> {
  storageKey: string;
  defaults: () => T;
  /** Optional custom merge for nested sub-objects. Default: shallow `{ ...defaults, ...parsed }`. */
  merge?: (defaults: T, parsed: Partial<T>) => T;
}

export class CalculatorStore<T> {
  current: T;                 // $state-wrapped reactive value
  lastEdited: LastEdited;     // stamped by persist(), cleared by reset()

  constructor(options: CalculatorStoreOptions<T>);
  init(): void;               // eager-called by constructor; safe to call repeatedly
  persist(): void;            // writes localStorage[storageKey] + lastEdited.stamp()
  reset(): void;              // clears storage + ts entry + resets to defaults()
}
```

The constructor automatically:
- Creates `lastEdited = new LastEdited(`${options.storageKey}_ts`)` — so passing `storageKey: 'nicu_uac_uvc_state'` reproduces the existing `nicu_uac_uvc_state_ts` timestamp key.
- Calls `init()` eagerly — no need for the consuming module or route to call `init()` itself.
- Falls back to a shallow `{ ...defaults(), ...parsed }` merge when `options.merge` is omitted — sufficient for `UacUvcStateData` because there are no nested sub-objects.
