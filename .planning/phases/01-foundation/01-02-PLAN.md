Replace src/app.html with the FOUC-safe template documented in RESEARCH.md Pattern 3. The inline script MUST be the very first element in `<head>` — before any `<link>` tags — so it runs synchronously before CSS is parsed:

```html
<!doctype html>
<html lang="en">
  <head>
    <meta charset="utf-8" />
    <!-- DS-03 / D-06: FOUC prevention — MUST be first in head, before any stylesheet -->
    <!-- Reads persisted theme from localStorage and applies .dark class + data-theme -->
    <!-- before first paint so Tailwind dark: variants activate instantly -->
    <script>
      (function () {
        var stored = localStorage.getItem('nicu_assistant_theme');
        var prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
        var isDark = stored ? stored === 'dark' : prefersDark;
        if (isDark) document.documentElement.classList.add('dark');
        document.documentElement.setAttribute('data-theme', isDark ? 'dark' : 'light');
      })();
    </script>
    <!-- DS-04: Plus Jakarta Sans — Google Fonts with display=swap -->
    <link rel="preconnect" href="https://fonts.googleapis.com" />
    <link rel="preconnect" href="https://fonts.gstatic.com" crossorigin />
    <link
      href="https://fonts.googleapis.com/css2?family=Plus+Jakarta+Sans:wght@400;500;600;700;800&display=swap"
      rel="stylesheet"
    />
    <!-- NAV-04: viewport-fit=cover required for iOS safe-area-inset in standalone PWA mode -->
    <meta name="viewport" content="width=device-width, initial-scale=1, viewport-fit=cover" />
    <link rel="icon" href="%sveltekit.assets%/favicon.png" />
    %sveltekit.head%
  </head>
  <body data-sveltekit-preload-data="hover">
    <div style="display: contents">%sveltekit.body%</div>
  </body>
</html>
```

Note: Do NOT add `<link rel="manifest">` yet — that is Phase 4 (PWA). Do NOT add a CSP meta tag — that can interfere with SvelteKit's inline scripts during development.

/* DS-02 / D-06: Class-based dark variant for Tailwind CSS 4 -->
/* Tailwind CSS 4 removes the darkMode config option; use @custom-variant instead. -->
/* Both .dark class AND [data-theme="dark"] attribute activate dark mode. -->
/* The FOUC inline script in app.html sets both on <html> before first paint. -->
@custom-variant dark (&:where(.dark, .dark *));

@theme {

  --color-clinical-50:  oklch(97.5% 0.018 230);
  --color-clinical-100: oklch(93% 0.04 225);
  --color-clinical-200: oklch(86% 0.08 220);
  --color-clinical-300: oklch(77% 0.13 225);
  --color-clinical-400: oklch(67% 0.17 210);
  --color-clinical-500: oklch(59% 0.19 210);
  --color-clinical-600: oklch(49% 0.17 220);
  --color-clinical-700: oklch(40% 0.14 225);
  --color-clinical-800: oklch(30% 0.11 230);
  --color-clinical-900: oklch(22% 0.08 235);

  --color-bmf-500: oklch(72% 0.18 65);
  --color-bmf-600: oklch(62% 0.16 60);
  --color-bmf-700: oklch(50% 0.13 55);
  --color-bmf-800: oklch(38% 0.10 50);
  --color-bmf-900: oklch(27% 0.07 48);

  --color-surface:      var(--color-surface);
  --color-surface-alt:  var(--color-surface-alt);
  --color-surface-card: var(--color-surface-card);
  --color-border:       var(--color-border);
  --color-text-primary:    var(--color-text-primary);
  --color-text-secondary:  var(--color-text-secondary);
  --color-text-tertiary:   var(--color-text-tertiary);
  --color-accent:       var(--color-accent);
  --color-accent-light: var(--color-accent-light);
  --color-error:        var(--color-error);
  --color-error-light:  var(--color-error-light);

  /* DS-04: Font stack — Plus Jakarta Sans with system fallbacks */
  --font-sans: "Plus Jakarta Sans", ui-sans-serif, system-ui, sans-serif;

  /* Custom type scale for clinical UI (fills Tailwind gaps) */
  --text-2xs:     0.6875rem;   /* 11px — minimum readable labels */
