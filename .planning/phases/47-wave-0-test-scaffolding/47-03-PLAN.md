```ts
// e2e/webkit-smoke.spec.ts
// Phase 47 / TEST-03: smoke check that the webkit-iphone Playwright project
// is wired correctly. Behavioral assertions belong to Phase 49 (DRAWER-TEST-03).
//
// Goal: prove (a) the project loads, (b) the page navigates, (c) WebKit reports
// a non-null window.visualViewport (real WebKit, independent of the jsdom polyfill
// from Plan 47-01).
import { test, expect } from '@playwright/test';

test('webkit-iphone project is wired and visualViewport is defined', async ({ page }) => {
  await page.goto('/');
  const hasVV = await page.evaluate(
    () => typeof window.visualViewport === 'object' && window.visualViewport !== null
  );
  expect(hasVV).toBe(true);
});

test('webkit-iphone project delivers iPhone user-agent and 393x660 viewport', async ({ page }) => {
  await page.goto('/');
  // Sanity-check the device descriptor so a future config regression that
  // swapped the device (e.g. 'Pixel 5' or 'Desktop Safari') would surface here.
  const ua = await page.evaluate(() => navigator.userAgent);
  expect(ua).toContain('iPhone');
  expect(page.viewportSize()).toEqual({ width: 393, height: 660 });
});
```

Constraints:
- Use bare `await page.goto('/')` — do NOT include the disclaimer-dismiss boilerplate from `pwa.spec.ts:5-9` (the smoke spec doesn't interact with calculator UI).
- Do NOT wrap in a `test.describe(...)` block — single file, single concern, two tests is below the threshold for grouping (PATTERNS.md §`e2e/webkit-smoke.spec.ts`).
- Do NOT add `test.use({ ... })` to override the project's device — the project IS the device.
- Do NOT add `test.skip(...)` based on `browserName` — both tests should run only under `webkit-iphone` AND under `chromium` (and pass under both, since chromium ALSO defines `window.visualViewport` and the second test's sanity assertions match `Desktop Chrome` poorly — see next bullet).

**Important caveat:** the second test (`'iPhone user-agent and 393x660 viewport'`) WILL fail under the `chromium` project because `Desktop Chrome` reports a desktop user-agent and a 1280×720 viewport (not 393×660 iPhone). To preserve D-13 / D-18 (existing 99-passing chromium suite must not regress), this second test MUST be limited to `webkit-iphone`. Use the `test.skip` pattern keyed on `browserName`:

```ts
test('webkit-iphone project delivers iPhone user-agent and 393x660 viewport', async ({ page, browserName }) => {
  test.skip(browserName !== 'webkit', 'webkit-iphone-only sanity check');
  await page.goto('/');
  const ua = await page.evaluate(() => navigator.userAgent);
  expect(ua).toContain('iPhone');
  expect(page.viewportSize()).toEqual({ width: 393, height: 660 });
});
```

| Boundary | Description |
|----------|-------------|

| Threat ID | Category | Component | Disposition | Mitigation Plan |
