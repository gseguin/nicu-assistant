```typescript
import type { FeedsInputRanges, FrequencyOption, CadenceOption, Advisory } from './types.js';
import config from './feeds-config.json';

export const defaults = config.defaults as {
  weightKg: number;
  trophicMlKgDay: number;
  advanceMlKgDay: number;
  goalMlKgDay: number;
  trophicFrequency: string;
  advanceCadence: string;
  totalFluidsMlHr: number;
  tpnDex1Pct: number;
  tpnMl1Hr: number;
  tpnDex2Pct: number;
  tpnMl2Hr: number;
  smofMl: number;
  enteralMl: number;
  enteralKcalPerOz: number;
};

export const inputs: FeedsInputRanges = config.inputs as FeedsInputRanges;
export const frequencyOptions: FrequencyOption[] = config.dropdowns.frequency as FrequencyOption[];
export const cadenceOptions: CadenceOption[] = config.dropdowns.cadence as CadenceOption[];
export const advisories: Advisory[] = config.advisories as Advisory[];

export function getFrequencyById(id: string): FrequencyOption | undefined {
  return frequencyOptions.find((f) => f.id === id);
}

export function getCadenceById(id: string): CadenceOption | undefined {
  return cadenceOptions.find((c) => c.id === id);
}

/**
 * Resolve advance events per day from cadence option and feeds per day.
 * - Relative cadences (every, every-other, every-3rd): feedsPerDay / value
 * - Absolute cadences (bid, qd): value directly
 */
export function resolveAdvanceEventsPerDay(cadence: CadenceOption, feedsPerDay: number): number {
  return cadence.type === 'absolute' ? cadence.value : feedsPerDay / cadence.value;
}
```

```typescript
/** kcal per gram of dextrose (Atwater factor for glucose monohydrate). */
export const DEXTROSE_KCAL_PER_GRAM = 3.4;

/** kcal per ml of SMOF 20% lipid emulsion (per xlsx Sheet1 B15 formula: =B7*2). */
export const LIPID_KCAL_PER_ML = 2;

/** ml per fluid ounce — xlsx Sheet1 uses 30, not 29.5735. */
export const ML_PER_OZ = 30;

/** Hours per day — for ml/hr to ml/day conversions. */
export const HOURS_PER_DAY = 24;
```

**Bedside advancement functions** (Sheet2 formulas):

1. `calculateTrophicMlPerFeed(weightKg, trophicMlKgDay, feedsPerDay)`:
   - Formula: `weightKg * trophicMlKgDay / feedsPerDay`
   - JSDoc: "xlsx Sheet2 C3: =C2*B3/6. The /6 is hardcoded q4h; we parameterize via feedsPerDay."

2. `calculateAdvanceStepMlPerFeed(weightKg, advanceMlKgDay, feedsPerDay, advanceEventsPerDay)`:
   - Formula: `weightKg * advanceMlKgDay / feedsPerDay / advanceEventsPerDay` (per D-06/FREQ-04)
   - JSDoc: "xlsx Sheet2 C4: =C2*B4/8/2. The /8 is feeds_per_day, /2 is bid cadence; we parameterize both."

3. `calculateGoalMlPerFeed(weightKg, goalMlKgDay, feedsPerDay)`:
   - Formula: `weightKg * goalMlKgDay / feedsPerDay`
   - JSDoc: "xlsx Sheet2 C5: =C2*B5/8."

4. `calculateTotalFluidsMlHr(weightKg, goalMlKgDay)`:
   - Formula: `weightKg * goalMlKgDay / HOURS_PER_DAY`
   - JSDoc: "total daily volume (weight*goal) converted to ml/hr."

5. `calculateIvBackfillRate(totalFluidsMlHr, enteralMlPerFeed, feedHours)`:
   - Formula: `totalFluidsMlHr - (enteralMlPerFeed / feedHours)`
   - JSDoc: "xlsx Sheet2 B9: =B7-(B8/3). The /3 is q3h feed interval in hours; we parameterize."

6. `calculateBedsideAdvance(weightKg, trophicMlKgDay, advanceMlKgDay, goalMlKgDay, feedsPerDay, advanceEventsPerDay)`:
   - Returns a `BedsideResult` object combining all individual calculations above.
   - Includes `trophicMlKgDay`, `advanceMlKgDay`, `goalMlKgDay` echo-back in result for CORE-06.
   - Computes `totalFluidsMlHr` from weight and goal.

**Full nutrition functions** (Sheet1 formulas):

7. `calculateDextroseKcal(dex1Pct, ml1, dex2Pct, ml2)`:
   - Formula: `((dex1Pct / 100) * ml1 + (dex2Pct / 100) * ml2) * DEXTROSE_KCAL_PER_GRAM` (per D-04)
   - JSDoc: "xlsx Sheet1 B14: =((B3/100*B4)+(B5/100*B6))*3.4. Sums BOTH TPN lines."

8. `calculateLipidKcal(smofMl)`:
   - Formula: `smofMl * LIPID_KCAL_PER_ML`
   - JSDoc: "xlsx Sheet1 B15: =B7*2"

9. `calculateEnteralKcal(enteralMl, enteralKcalPerOz)`:
   - Formula: `(enteralMl * enteralKcalPerOz) / ML_PER_OZ`
   - JSDoc: "xlsx Sheet1 B16: =(B10*B9)/30"

10. `calculateFullNutrition(weightKg, dex1Pct, ml1, dex2Pct, ml2, smofMl, enteralMl, enteralKcalPerOz)`:
    - Returns `FullNutritionResult` with:
      - `dextroseKcal` from calculateDextroseKcal
      - `lipidKcal` from calculateLipidKcal
      - `enteralKcal` from calculateEnteralKcal
      - `totalKcal = dextroseKcal + lipidKcal + enteralKcal`
      - `totalKcalPerKgDay = totalKcal / weightKg`
      - `mlPerKg = (ml1 + ml2 + enteralMl) / weightKg` (xlsx B17: =(B4+B6+B10)/B13)
      - `autoAdvanceMlPerFeed = weightKg * 30 / 8 / 2` (xlsx B20: =B13*30/8/2 — hardcoded to q3h/bid)
    - JSDoc: cite xlsx Sheet1 B14-B18 + B20 formulas.

**Advisory checking** (per D-09):

11. `checkAdvisories(inputs, results, mode, advisories)`:
