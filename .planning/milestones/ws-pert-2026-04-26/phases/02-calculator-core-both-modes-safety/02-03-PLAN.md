**Concrete script body (paste verbatim, adjusting only formatting):**

```svelte
<script lang="ts">
  import { pertState } from './state.svelte.js';
  import {
    inputs,
    medications,
    formulas,
    getStrengthsForMedication
  } from './config.js';
  import type { PertMode } from './types.js';
  import type { SelectOption } from '$lib/shared/types.js';
  import SegmentedToggle from '$lib/shared/components/SegmentedToggle.svelte';
  import RangedNumericInput from '$lib/shared/components/RangedNumericInput.svelte';
  import NumericInput from '$lib/shared/components/NumericInput.svelte';
  import SelectPicker from '$lib/shared/components/SelectPicker.svelte';

  // Persist on every state change. Defensive duplicate of the PertCalculator's
  // own $effect so the inputs work standalone when mounted only in the mobile
  // <InputDrawer> (PertCalculator may not be mounted in that path's React tree;
  //  in our case both are always mounted, but matching the FeedAdvanceInputs
  //  pattern keeps the mental model uniform).
  $effect(() => {
    JSON.stringify(pertState.current);
    pertState.persist();
  });

  // Mode toggle options (D-06)
  const modeOptions: { value: PertMode; label: string }[] = [
    { value: 'oral', label: 'Oral' },
    { value: 'tube-feed', label: 'Tube-Feed' }
  ];

  // Medication SelectPicker options (PERT-ORAL-04 / PERT-TUBE-05)
  const medicationOptions: SelectOption[] = medications.map((m) => ({
    value: m.id,
    label: m.brand
  }));

  // Formula SelectPicker options (PERT-TUBE-02 — 17 entries, search-enabled)
  const formulaOptions: SelectOption[] = formulas.map((f) => ({
    value: f.id,
    label: f.name
  }));

  // Strength SelectPicker options derived from selected medication (PERT-ORAL-05).
  // en-US locale formatting per UI-SPEC §Copywriting Contract: "12,000 units".
  // Empty array when no medication selected — Strength picker shows placeholder.
  let strengthOptions: SelectOption[] = $derived(
    pertState.current.medicationId
      ? getStrengthsForMedication(pertState.current.medicationId).map((s) => ({
          value: String(s),
          label: `${s.toLocaleString('en-US')} units`
        }))
      : []
  );

  // SelectPicker bind:value requires string. Bridge state.strengthValue (number|null)
  // to a string-bound proxy with two-way sync.
  let strengthStr = $state<string>('');

  // Sync state.strengthValue (number|null) -> local string (when state changes externally,
  // e.g. medication change resets strength to null per D-11, or reset() clears all).
  $effect(() => {
    const next =
      pertState.current.strengthValue === null
        ? ''
        : String(pertState.current.strengthValue);
    if (next !== strengthStr) {
      strengthStr = next;
    }
  });

  // Sync local string -> state.strengthValue when SelectPicker writes.
  $effect(() => {
    if (strengthStr === '') {
      if (pertState.current.strengthValue !== null) {
        pertState.current.strengthValue = null;
      }
      return;
    }
    const parsed = parseInt(strengthStr, 10);
    if (Number.isFinite(parsed) && parsed !== pertState.current.strengthValue) {
      pertState.current.strengthValue = parsed;
    }
  });

  // SelectPicker bind:value requires string. Bridge formulaId (string|null) to a
  // string-bound proxy with two-way sync. Mirrors the strengthStr pattern above and
  // FeedAdvanceInputs.svelte:55-74. NOT an `as` cast (writable lvalue requirement).
  let formulaIdStr = $state<string>(pertState.current.tubeFeed.formulaId ?? '');

  // Sync state.tubeFeed.formulaId (string|null) -> local string (external resets, etc.)
  $effect(() => {
    const next = pertState.current.tubeFeed.formulaId ?? '';
    if (next !== formulaIdStr) {
      formulaIdStr = next;
    }
  });

  // Sync local string -> state.tubeFeed.formulaId when SelectPicker writes.
  $effect(() => {
    const next = formulaIdStr === '' ? null : formulaIdStr;
    if (next !== pertState.current.tubeFeed.formulaId) {
      pertState.current.tubeFeed.formulaId = next;
    }
  });

  // Same string-bridge pattern for medicationId. Cannot use `as string` cast inside
  // bind:value (writable lvalue requirement); use $state proxy + two $effect blocks.
  let medicationIdStr = $state<string>(pertState.current.medicationId ?? '');

  // Sync state.medicationId (string|null) -> local string.
  $effect(() => {
    const next = pertState.current.medicationId ?? '';
    if (next !== medicationIdStr) {
      medicationIdStr = next;
    }
  });

  // Sync local string -> state.medicationId when SelectPicker writes.
  $effect(() => {
    const next = medicationIdStr === '' ? null : medicationIdStr;
    if (next !== pertState.current.medicationId) {
      pertState.current.medicationId = next;
    }
  });

  // D-11: medication change resets strengthValue to null. Track last seen
  // medicationId; on change, force-clear strengthValue (which propagates
  // through the strengthStr $effect above to clear the SelectPicker).
  let lastMedId = $state<string | null>(pertState.current.medicationId);
  $effect(() => {
    const cur = pertState.current.medicationId;
    if (cur !== lastMedId) {
      lastMedId = cur;
      // Reset only if it's a genuine medication change (not the initial mount).
      // The strengthValue may have been restored from localStorage with a stale
      // strength for the new medication; clear it.
      pertState.current.strengthValue = null;
    }
  });
</script>
```
